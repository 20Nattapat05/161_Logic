1. เซต

A = {x∈S | -2≤x≤2} , B = {x∈S | -1<x<3}

เมื่อ S = Z
  • A∪B = {-2,-1,0,1,2}
  • A∩B = {0,1,2}
  • A-B = {-2,-1}
  • Aᶜ = Z - {-2,-1,0,1,2}
  • A ∪ Bᶜ = Z
  • (A∪B)ᶜ = Z - {-2,-1,0,1,2}
เมื่อ S = R
  • A∪B = [-2,3)
  • A∩B = (-1,2]
  • A-B = [-2,-1]
  • Aᶜ = (-∞,-2) ∪ (2,∞)
  • A ∪ Bᶜ = R
  • (A∪B)ᶜ = (-∞,-2) ∪ [3,∞)
